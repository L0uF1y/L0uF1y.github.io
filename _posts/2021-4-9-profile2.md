* 折半插入排序 O(n^2)

```go
func RhInsert(array *[] int) []int{
	length := len(*array)
	for i :=1;i < length;i++{
		if (*array)[i] < (*array)[i-1]{
			low := 0
			high := i - 1
			tmp :=(*array)[i]
			for{
				if low > high{
					break
				}
				mid := (low + high)/2
				if (*array)[mid] > tmp{
					high = mid - 1
				}else{
					low = mid + 1
				}
			}
			for j := i;j > low;j--{
				(*array)[j] = (*array)[j-1]
			}
			(*array)[low] = tmp
		}
	}
	return *array
}
```
* 选择排序 O(n^2)
```go
func ChooseSort(a *[]int) []int{
	length := len(*a)
	for i:=0;i < length;i++{
		min :=i
		for j :=i+1;j < length;j++{
			if (*a)[j] < (*a)[min]{
				min = j
			}
		}
		if min != i{
			(*a)[i],(*a)[min] = (*a)[min],(*a)[i]
		}
	}
	return *a
}
```
